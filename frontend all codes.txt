        function switchForm(type) {
            const slider = document.getElementById('slider');
            const loginSection = document.getElementById('login-section');
            const registerSection = document.getElementById('register-section');
            const headerTitle = document.getElementById('header-title');
            const tabs = document.querySelectorAll('.tab');

            if (type === 'register') {
                slider.style.transform = 'translateX(100%)';
                loginSection.classList.remove('active');
                registerSection.classList.add('active');
                headerTitle.innerText = "Join Finerva";
                tabs[1].classList.add('active');
                tabs[0].classList.remove('active');
            } else {
                slider.style.transform = 'translateX(0)';
                registerSection.classList.remove('active');
                loginSection.classList.add('active');
                headerTitle.innerText = "Welcome Back";
                tabs[0].classList.add('active');
                tabs[1].classList.remove('active');
            }
        }

let stockData = null;

function analyze() {
    const ticker = document.getElementById('ticker').value.toUpperCase();
    if (!ticker) {
        alert('Please enter a stock ticker');
        return;
    }

    // Generate random data
    const basePrice = 100 + Math.random() * 100;
    const change = (Math.random() * 10 - 4).toFixed(2);
    const isPositive = parseFloat(change) >= 0;
    const finalPrice = basePrice * (1 + parseFloat(change) / 100);
    const high = finalPrice * (1 + Math.random() * 0.05);
    const low = finalPrice * (1 - Math.random() * 0.05);
    const volume = (Math.random() * 10 + 1).toFixed(1) + 'M';

    stockData = {
        ticker: ticker,
        open: basePrice.toFixed(2),
        high: high.toFixed(2),
        low: low.toFixed(2),
        close: finalPrice.toFixed(2),
        change: change,
        isPositive: isPositive,
        volume: volume
    };

    // Update UI
    document.getElementById('stockName').innerText = ticker;
    document.getElementById('tickerDisplay').innerText = ticker + ' Stock';
    document.getElementById('plAmount').innerText = (isPositive ? '+' : '') + '$' + (finalPrice - basePrice).toFixed(2);
    document.getElementById('plPercent').innerText = (isPositive ? '+' : '') + change + '%';
    
    const profitLossEl = document.getElementById('profitLoss');

function generateChart() {
    const ctx = document.getElementById('stockChart').getContext('2d');
    
    if (chartInstance) {
        chartInstance.destroy();
    }

    const isPositive = stockData.isPositive;
    const color = isPositive ? '#22c55e' : '#ef4444';
    const bgColor = isPositive ? 'rgba(34,197,94,0.2)' : 'rgba(239,68,68,0.2)';

    // Generate data points
    let dataPoints = [];
    let base = parseFloat(stockData.open);
    for (let i = 0; i < 50; i++) {
        base += (Math.random() - 0.5) * 2;
        dataPoints.push(base);
    }
    dataPoints.push(parseFloat(stockData.close));

    const labels = Array.from({length: dataPoints.length}, (_, i) => i + 1);

    if (currentChartType === 'line') {
        chartInstance = new Chart(ctx, {
            type: 'line',
            data: {
                labels: labels,
                datasets: [{
                    data: dataPoints,
                    borderColor: color,
                    backgroundColor: bgColor,
                    fill: true,
                    tension: 0.4,
                    pointRadius: 0,
                    pointHoverRadius: 6,
                    pointHoverBackgroundColor: color
                }]
            },
            options: {
                responsive: true,
                maintainAspectRatio: false,
                plugins: {
                    legend: { display: false }
                },
                scales: {
                    x: { display: false },
                    y: { 
                        display: true,
                        grid: { color: 'rgba(255,255,255,0.1)' },
                        ticks: { color: 'rgba(255,255,255,0.6)' }
                    }
                },
                interaction: {
                    intersect: false,
                    mode: 'index'
                }
            }
        });
    } else if (currentChartType === 'candle') {
        // Generate candle data
        const candleData = [];
        let prevClose = parseFloat(stockData.open);
        for (let i = 0; i < dataPoints.length; i++) {
            const open = prevClose;
            const close = dataPoints[i];
            const high = Math.max(open, close) + Math.random() * 2;
            const low = Math.min(open, close) - Math.random() * 2;
            candleData.push({ x: i + 1, o: open, h: high, l: low, c: close });
            prevClose = close;
        }

        chartInstance = new Chart(ctx, {
            type: 'bar',
            data: {
                labels: labels,
                datasets: [{
                    data: candleData.map(d => [d.o, d.h, d.l, d.c]),
                    backgroundColor: candleData.map(d => d.c >= d.o ? '#22c55e' : '#ef4444'),
                    borderColor: candleData.map(d => d.c >= d.o ? '#22c55e' : '#ef4444'),
                    borderWidth: 1
                }]
            },
            options: {
                responsive: true,
                maintainAspectRatio: false,
                plugins: {
                    legend: { display: false },
                    tooltip: {
                        callbacks: {
                            label: function(context) {
                                const d = candleData[context.dataIndex];
                                return [
                                    'Open: $' + d.o.toFixed(2),
                                    'High: $' + d.h.toFixed(2),
                                    'Low: $' + d.l.toFixed(2),
                                    'Close: $' + d.c.toFixed(2)
                                ];
                            }
                        }
                    }
                },
                scales: {
                    x: { display: false },
                    y: { 
                        display: true,
                        grid: { color: 'rgba(255,255,255,0.1)' },
                        ticks: { color: 'rgba(255,255,255,0.6)' }
                    }
                }
            }
        });
    } else if (currentChartType === 'tally') {
        // Tally chart - vertical bars
        const tallyData = dataPoints.map((v, i) => ({
            x: i + 1,
            y: v
        }));

        chartInstance = new Chart(ctx, {
            type: 'bar',
            data: {
                labels: labels,
                datasets: [{
                    data: tallyData,
                    backgroundColor: tallyData.map(d => d.y >= parseFloat(stockData.open) ? '#22c55e' : '#ef4444'),
                    borderRadius: 4
                }]
            },
            options: {
                responsive: true,
                maintainAspectRatio: false,
                plugins: { legend: { display: false } },
                scales: {
                    x: { display: false },
                    y: { 
                        display: true,
                        grid: { color: 'rgba(255,255,255,0.1)' },
                        ticks: { color: 'rgba(255,255,255,0.6)' }
                    }
                }
            }
        });
    } else if (currentChartType === 'renko') {
        // Renko chart - simplified brick representation
        const renkoData = [];
        let renkoBase = parseFloat(stockData.open);
        const brickSize = Math.abs(parseFloat(stockData.close) - parseFloat(stockData.open)) / 10;
        
        for (let i = 0; i < 20; i++) {
            const change = (Math.random() - 0.5) * 2 * brickSize;
            renkoBase += change;
            renkoData.push({
                x: i + 1,
                y: renkoBase
            });
        }

        chartInstance = new Chart(ctx, {
            type: 'line',
            data: {
                labels: Array.from({length: 20}, (_, i) => i + 1),
                datasets: [{
                    data: renkoData.map(d => d.y),
                    borderColor: color,
                    backgroundColor: bgColor,
                    fill: true,
                    tension: 0,
                    pointRadius: 8,
                    pointStyle: 'rectRot',
                    pointBackgroundColor: renkoData.map(d => d.y >= parseFloat(stockData.open) ? '#22c55e' : '#ef4444')
                }]
            },
            options: {
                responsive: true,
                maintainAspectRatio: false,
                plugins: { legend: { display: false } },
                scales: {
                    x: { display: false },
                    y: { 
                        display: true,
                        grid: { color: 'rgba(255,255,255,0.1)' },
                        ticks: { color: 'rgba(255,255,255,0.6)' }
                    }
                }
            }
        });
    } else if (currentChartType === 'baseline') {
        // Baseline chart with area
        const baselineValue = parseFloat(stockData.open);
        
        chartInstance = new Chart(ctx, {
            type: 'line',
            data: {
                labels: labels,
                datasets: [
                    {
                        label: 'Price',
                        data: dataPoints,
                        borderColor: color,
                        backgroundColor: 'transparent',
                        tension: 0.4,
                        pointRadius: 0
                    },
                    {
                        label: 'Baseline',
                        data: Array(dataPoints.length).fill(baselineValue),
                        borderColor: '#f59e0b',
                        borderDash: [5, 5],
                        borderWidth: 2,
                        pointRadius: 0,
                        fill: false
                    }
                ]
            },
            options: {
                responsive: true,
                maintainAspectRatio: false,
                plugins: { legend: { display: false } },
                scales: {
                    x: { display: false },
                    y: { 
                        display: true,
                        grid: { color: 'rgba(255,255,255,0.1)' },
                        ticks: { color: 'rgba(255,255,255,0.6)' }
                    }
                }
            }
        });
    }
}

        let portfolio = [];

        function addAsset() {
            const symbol = document.getElementById('symbol').value.toUpperCase();
            const qty = parseFloat(document.getElementById('qty').value);
            const buyPrice = parseFloat(document.getElementById('price').value);

            if (!symbol || isNaN(qty) || isNaN(buyPrice)) {
                alert('Please fill in all fields correctly');
                return;
            }

            // Generate current price with random variation
            const changePercent = (Math.random() - 0.4) * 0.4; // -20% to +20%
            const currentPrice = buyPrice * (1 + changePercent);
            const plPercent = ((currentPrice - buyPrice) / buyPrice) * 100;
            const plAmount = (currentPrice - buyPrice) * qty;

            portfolio.push({
                symbol,
                qty,
                buyPrice,
                currentPrice,
                plPercent,
                plAmount
            });

            updateDisplay();
            
            // Clear inputs
            document.getElementById('symbol').value = '';
            document.getElementById('qty').value = '';
            document.getElementById('price').value = '';
        }

        function removeAsset(index) {
            portfolio.splice(index, 1);
            updateDisplay();
        }

        function updateDisplay() {
            const body = document.getElementById('portfolio-body');
            let totalInvested = 0, currentTotalVal = 0;

            if (portfolio.length === 0) {
                body.innerHTML = `
                    <tr>
                        <td colspan="6" class="empty-state">
                            <svg viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2">
                                <rect x="2" y="7" width="20" height="14" rx="2" ry="2"></rect>
                                <path d="M16 21V5a2 2 0 0 0-2-2h-4a2 2 0 0 0-2 2v16"></path>
                            </svg>
                            <p>No assets in your portfolio yet.<br>Add your first asset above!</p>
                        </td>
                    </tr>`;
                
                // Update summary
                document.getElementById('totalInv').innerText = '$0.00';
                document.getElementById('currentVal').innerText = '$0.00';
                document.getElementById('totalPL').innerText = '0.00%';
                document.getElementById('totalPL').className = 'value';
                document.getElementById('assetCount').innerText = '0';
                
                // Update allocation
                document.getElementById('allocationFill').style.width = '0%';
                document.getElementById('allocationList').innerHTML = '<div class="allocation-item"><span class="allocation-label">No assets yet</span><span class="allocation-value">-</span></div>';
                
                // Disable download button
                document.getElementById('downloadBtn').disabled = true;
                return;
            }

            body.innerHTML = '';
            const colors = ['#00aeef', '#22c55e', '#f59e0b', '#ef4444', '#8b5cf6', '#ec4899'];

            portfolio.forEach((item, index) => {
                totalInvested += (item.qty * item.buyPrice);
                currentTotalVal += (item.qty * item.currentPrice);
                
                const plClass = item.plPercent >= 0 ? 'profit' : 'loss';
                
                body.innerHTML += `
                    <tr>
                        <td style="font-weight: 900; color: #002d5b;">${item.symbol}</td>
                        <td>${item.qty}</td>
                        <td>$${item.buyPrice.toFixed(2)}</td>
                        <td>$${item.currentPrice.toFixed(2)}</td>
                        <td class="${plClass}">${item.plPercent >= 0 ? '+' : ''}${item.plPercent.toFixed(2)}%</td>
                        <td><button class="remove-btn" onclick="removeAsset(${index})">Remove</button></td>
                    </tr>`;
            });

            // Update summary
            document.getElementById('totalInv').innerText = '$' + totalInvested.toLocaleString();
            document.getElementById('currentVal').innerText = '$' + currentTotalVal.toLocaleString();
            
            const totalPL = totalInvested > 0 ? ((currentTotalVal - totalInvested) / totalInvested) * 100 : 0;
            const plEl = document.getElementById('totalPL');
            plEl.innerText = (totalPL >= 0 ? '+' : '') + totalPL.toFixed(2) + '%';
            plEl.className = 'value ' + (totalPL >= 0 ? 'positive' : 'negative');
            
            document.getElementById('assetCount').innerText = portfolio.length;

            // Update allocation
            const allocationList = document.getElementById('allocationList');
            allocationList.innerHTML = '';
            
            portfolio.forEach((item, index) => {
                const percent = ((item.qty * item.currentPrice) / currentTotalVal) * 100;
                const color = colors[index % colors.length];
                allocationList.innerHTML += `
                    <div class="allocation-item">
                        <span class="allocation-label" style="color: ${color}">${item.symbol}</span>
                        <span class="allocation-value">${percent.toFixed(1)}%</span>
                    </div>`;
            });

            // Enable download button
            document.getElementById('downloadBtn').disabled = false;
        }

<script>
    const canvas = document.getElementById('chat-canvas');
    let questionsCount = 0;
    let topicsCount = 0;
    const topicsSet = new Set();
    let sessionStartTime = Date.now();

    // Update session time
    setInterval(() => {
        const elapsed = Math.floor((Date.now() - sessionStartTime) / 1000);
        const minutes = Math.floor(elapsed / 60);
        const seconds = elapsed % 60;
        document.getElementById('sessionTime').textContent = `${minutes}:${seconds.toString().padStart(2, '0')}`;
    }, 1000);

    function handleKeyPress(event) {
        if (event.key === 'Enter') {
            processInput();
        }
    }

    function handlePill(txt) {
        addUserMsg(txt);
        questionsCount++;
        topicsSet.add(txt.split(' ')[0]);
        topicsCount = topicsSet.size;
        updateStats();
        setTimeout(() => botResponse(txt), 700);
    }

    function processInput() {
        const input = document.getElementById('chatInput');
        if(!input.value.trim()) return;
        addUserMsg(input.value);
        questionsCount++;
        topicsSet.add(input.value.split(' ')[0]);
        topicsCount = topicsSet.size;
        updateStats();
        const val = input.value;
        input.value = "";
        setTimeout(() => botResponse(val), 700);
    }

    function updateStats() {
        document.getElementById('questionsCount').textContent = questionsCount;
        document.getElementById('topicsCount').textContent = topicsCount;
    }

    function addUserMsg(msg) {
        const div = document.createElement('div');
        div.className = 'msg msg-user';
        div.textContent = msg;
        canvas.appendChild(div);
        canvas.scrollTop = canvas.scrollHeight;
    }

    function botResponse(input) {
        const query = input.toLowerCase();
        let reply = "";

        if(query.includes("tax")) {
            reply = "To minimize tax, prioritize <strong>Tax-Saving Mutual Funds (ELSS)</strong>. Since your Portfolio is already 60% Tech, ELSS helps you diversify while lowering your taxable income.";
        } 
        else if(query.includes("inflation")) {
            reply = "Inflation is the 'Hidden Tax' on your cash. With current rates at 5.5%, your savings account is likely losing value. Our <strong>Stock Predictor</strong> shows assets that historically outpace inflation.";
        }
        else if(query.includes("house") || query.includes("real estate")) {
            reply = "Buying a house is an 'Illiquid' asset. Based on your <strong>Portfolio Analyzer</strong>, you have high liquidity. A Home Loan might be better than a lump-sum payment to keep your capital growing in stocks.";
        }
        else if(query.includes("retire")) {
            reply = "For a comfortable retirement, aim for the <strong>25X Rule</strong>: Save 25 times your annual expenses. I can calculate your 'FIRE' number if you provide your monthly spending.";
        }
        else if(query.includes("tech") || query.includes("portfolio")) {
            reply = "Your portfolio has a high concentration in Technology. While our <strong>News Intelligence</strong> is bullish on AI, I suggest shifting 10% to Debt or Gold to hedge against a sector correction.";
        }
        else if(query.includes("invest") || query.includes("strategy")) {
            reply = "For optimal returns, consider a <strong>diversified portfolio</strong>: 60% equities, 25% bonds, 10% gold, 5% cash. This balances growth potential with risk management.";
        }
        else if(query.includes("save") || query.includes("saving")) {
            reply = "Start with the <strong>50/30/20 rule</strong>: 50% needs, 30% wants, 20% savings. Use automatic transfers to make saving consistent.";
        }
        else {
            reply = "That's a great financial question! In general, a <strong>Balanced Asset Allocation</strong> is your best defense against market volatility. Would you like me to elaborate on any specific topic?";
        }

        const div = document.createElement('div');
        div.className = 'msg msg-bot';
        div.innerHTML = reply;
        canvas.appendChild(div);
        canvas.scrollTop = canvas.scrollHeight;
    }
